axios

安装axios，这里使用的cnpm

    npm isntall axios --save

在react-app中，引用axios

    import axios from "axios";
![import](png/import.png) #### 简单使用 
    getDate() {
        const _this = this;
        axios
          .get("url")
          .then(res => {
            _this.setState({ data: res.data });
          });
    }

    componentDidMount() {
        this.getDate();
    }

ajax

安装jquery，这里使用的cnpm ，引用 $

    npm isntall jquery --save

    import $ from 'jquery';

简单使用

    componentDidMount() {
            this.listRender();
    }

    listRender = e => {
        $.ajax({
            type: "POST",
            url: "http://127.0.0.1:8360/goods/index",
            dataType: "json",
        }).then(res => {
        this.setState({
                list: res.data.data,
            });
        });
    }

fetch

    getDate() {
        fetch("../aboutdata.json", {
          method: "GET"
        })
          .then(res => res.json())
          .then(data => {
            this.setState({ aboutdata: data });
          });
    }

    componentDidMount() {
        this.getDate();
    }
